	\item From the textbook (Schaum's Outline of Beginning Linear Algebra): 11.57, 11.60 part a), 11.58.
	
	\item Eigenvalues: 
	\begin{enumerate}
	\item Find the eigenvalues only of the matrix
	$$ A = \begin{pmatrix} 1 & 1 \\ -1 & 1\end{pmatrix} . $$
	\item If a matrix is real and symmetric, that is it has the form
	$$ A =  \begin{pmatrix} a & b \\ b & c\end{pmatrix}$$
	for $a,b,c\in\mathbb{R}$, show that $A$ must have real eigenvalues. 
	\end{enumerate}
	
	
	\item Let $A$ be a matrix with eigenvalues $\lambda_1=1, \lambda_2=2,...,\lambda_n=n$ and associated eigenvectors $\vec{v}_1,\vec{v}_2,...,\vec{v}_n$. 
	Let $\vec{b}$ be a vector that can be written as
	$$ \vec{b} = c_1\vec{v}_1+c_2\vec{v}_2+\cdots +c_n\vec{v}_n.$$
	Find the constants $d_1,d_2,..,d_n$ such that
	$$ A\vec{b} = d_1\vec{v}_1+d_2\vec{v}_2+\cdots +d_n\vec{v}_n.$$
	
	\item Let $A$ be a $4\times 4$ matrix with eigenvalues $\lambda_1=1$, $\lambda_2=\lambda_3=\lambda_4=-10$ with eigenvectors $\vec{v}_1,\vec{v}_2,\vec{v}_3,$ and $\vec{v}_4$. Use a diagonalization of $A$ to find the eigenvalues of, $A^n$. (Hint: the same diagonalization matrix for $A$ diagonalizes $A^n$). What happens as $n\to\infty$?
\end{enumerate}
